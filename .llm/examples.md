# Shop Client - Code Examples

## Overview

This document provides comprehensive code examples for the `shop-client` library, designed to help LLMs understand common usage patterns and generate appropriate code.

## Basic Examples

### 1. Simple Product Fetching

```typescript
import { ShopClient } from 'shop-client';

async function fetchAllProducts() {
  const shop = new ShopClient('anuki.in');
  
  try {
    const products = await shop.products.all();
    
    if (!products) {
      console.log('No products found or store unavailable');
      return;
    }
    
    console.log(`Found ${products.length} products`);
    
    // Display first few products
    products.slice(0, 5).forEach(product => {
      console.log(`${product.title} - $${product.price / 100}`);
      console.log(`Available: ${product.available}`);
      console.log(`Vendor: ${product.vendor}`);
      console.log('---');
    });
  } catch (error) {
    console.error('Error fetching products:', error);
  }
}

fetchAllProducts();
```

### 2. Store Information Retrieval

```typescript
import { ShopClient } from 'shop-client';

async function getStoreDetails() {
  const shop = new ShopClient('anuki.in');
  
  const storeInfo = await shop.getInfo();
  
  if (!storeInfo) {
    console.log('Store information not available');
    return;
  }
  
  console.log('Store Information:');
  console.log(`Name: ${storeInfo.name}`);
  console.log(`Domain: ${storeInfo.domain}`);
  console.log(`Description: ${storeInfo.description}`);
  console.log(`Country: ${storeInfo.country || 'Unknown'}`);
  
  // Social links
  if (Object.keys(storeInfo.socialLinks).length > 0) {
    console.log('\nSocial Links:');
    Object.entries(storeInfo.socialLinks).forEach(([platform, url]) => {
      console.log(`${platform}: ${url}`);
    });
  }
  
  // Featured products
  console.log(`\nFeatured Products: ${storeInfo.showcase.products.length}`);
  console.log(`Featured Collections: ${storeInfo.showcase.collections.length}`);
}

getStoreDetails();
```

### 3. Product Search and Filtering

```typescript
import { ShopClient } from 'shop-client';

async function searchProducts(searchTerm: string) {
  const shop = new ShopClient('anuki.in');
  
  const products = await shop.products.all();
  
  if (!products) {
    console.log('No products available');
    return [];
  }
  
  // Search by title, description, or tags
  const results = products.filter(product => {
    const searchLower = searchTerm.toLowerCase();
    return (
      product.title.toLowerCase().includes(searchLower) ||
      product.description.toLowerCase().includes(searchLower) ||
      product.tags.some(tag => tag.toLowerCase().includes(searchLower)) ||
      product.vendor.toLowerCase().includes(searchLower)
    );
  });
  
  console.log(`Found ${results.length} products matching "${searchTerm}"`);
  
  results.forEach(product => {
    console.log(`${product.title} - $${product.price / 100}`);
    console.log(`Tags: ${product.tags.join(', ')}`);
    console.log('---');
  });
  
  return results;
}

// Usage
searchProducts('shirt');
const searchResults = await searchProducts(shop, 'shirt');
console.log(`Found ${searchResults.length} products matching "shirt"`);
```

### Price Range Filtering

```typescript
async function getProductsByPriceRange(shop: ShopClient, minPrice: number, maxPrice: number) {
  const products = await shop.products.filter({
    minPrice,
    maxPrice,
    available: true
  });
  
  return products.sort((a, b) => a.price - b.price);
}

// Usage
const affordableProducts = await getProductsByPriceRange(shop, 10, 50);
console.log(`Found ${affordableProducts.length} products between $10-$50`);
```
